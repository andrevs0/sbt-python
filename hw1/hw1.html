class Cache(dict):
    def __init__(self, f):
        self.f = f
    
    def __call__(self, *args, **kwargs):
        key = tuple(list(args) + ['|'] + list(kwargs.items()))
        if key not in self:
            self[key] = self.f(*args, **kwargs)
        return self[key]

        return self.__im

    def __repr__(self):
